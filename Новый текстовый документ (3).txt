4.2 Настройка коммутатора Switch-HQ
На отдельной ВМ (Debian 12) выполнить:

bash
apt update && apt install vlan bridge-utils -y
modprobe 8021q
echo "8021q" >> /etc/modules

# Создаём мост с VLAN-фильтрацией
ip link add name br0 type bridge vlan_filtering 1
ip link set br0 up

# Привязываем физические порты (замените ens6-ens9 на свои интерфейсы)
ip link set ens6 master br0   # trunk к HQ-RTR
ip link set ens7 master br0   # access VLAN100 (HQ-SRV)
ip link set ens8 master br0   # access VLAN200 (HQ-CLI)
ip link set ens9 master br0   # access VLAN999 (управление)

# Настройка trunk на ens6
bridge vlan add vid 100 dev ens6
bridge vlan add vid 200 dev ens6
bridge vlan add vid 999 dev ens6

# Настройка access портов
bridge vlan add vid 100 pvid untagged dev ens7
bridge vlan del vid 1 dev ens7
bridge vlan add vid 200 pvid untagged dev ens8
bridge vlan del vid 1 dev ens8
bridge vlan add vid 999 pvid untagged dev ens9
bridge vlan del vid 1 dev ens9

# Сохранение конфигурации (опционально)
cat > /etc/network/interfaces << EOF
auto lo
iface lo inet loopback

auto br0
iface br0 inet manual
    bridge_ports ens6 ens7 ens8 ens9
    bridge_vlan_aware yes
    up bridge vlan add vid 100 dev ens6
    up bridge vlan add vid 200 dev ens6
    up bridge vlan add vid 999 dev ens6
    up bridge vlan add vid 100 pvid untagged dev ens7
    up bridge vlan add vid 200 pvid untagged dev ens8
    up bridge vlan add vid 999 pvid untagged dev ens9
EOF
Проверка: с HQ-RTR пинг до HQ-SRV (192.168.100.2) должен работать.
